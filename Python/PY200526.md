### 1. 클래스 예제

```py
class Student:
    def __init__(self, name, height):
        self._name = name
        self._height = height

    def __str__(self):
        return 'Student Class Info : {}, {}'.format(self._name, self._height)

    def __ge__(self, x):
        print('called. >> __ge__ method')
        if self._height >= x._height:
            return True
        else:
            return False

    def __le__(self, x):
        print('called. >> __le__ method')
        if self._height <= x._height:
            return True
        else:
            return False

    def __sub__(self, x):
        print('called. >> __sub__ method')
        return self._height - x._height


s1 = Student('James', 181)
s2 = Student('Mie', 165)

# 매직메소드 출력

print(s1 >= s2)
print(s1 <= s2)
print(s1 - s2)
print(s2 - s1)
print(s1)
print(s2)

print()
print()
```

### 2. 클래스 예제

- 벡터(Vector)

```py
class Vector(object):
    def __init__(self, *args):
        '''
         Create a vctor , example : v = Vector(1,2)
        '''
        if len(args) == 0:
            self._x, self._y = 0, 0
        else:
            self._x, self._y = args

    def __repr__(self):
        '''Returns the vector infomations'''
        return 'Vector(%d, %d)' % (self._x, self._y)

    def __add__(self, other):
        '''returns the vector additon of self and other'''
        return Vector(self._x + other._x, self._y + other._y)

    def __mul__(self, y):
        return Vector(self._x * y, self._y * y)

    def __bool__(self):
        return bool(max(self._x, self._y))


# vector 인스턴스 생성
v1 = Vector(3, 5)
v2 = Vector(15, 20)
v3 = Vector()

# 매직 메소드 출력
print(Vector.__init__.__doc__)
print(Vector.__repr__.__doc__)
print(Vector.__add__.__doc__)
print(v1, v2, v3)
print(v1 + v2)
print(v1 * 4)
print(v2 * 10)
print(bool(v1), bool(v2))
print(bool(v3))

print()
print()
```
